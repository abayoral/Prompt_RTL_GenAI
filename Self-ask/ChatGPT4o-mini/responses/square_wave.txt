```verilog
module square_wave (
    input wire clk,
    input wire [7:0] freq,
    output reg wave_out
);
    reg [7:0] count;

    always @(posedge clk) begin
        if (count < freq - 1) begin
            count <= count + 1;
        end else begin
            count <= 0;
            wave_out <= ~wave_out; // Toggle the output
        end
    end

    // Initialize wave_out
    initial begin
        wave_out = 0;
        count = 0;
    end

endmodule
```
