module comparator_4bit(
    input  [3:0] A,
    input  [3:0] B,
    output A_greater,
    output A_equal,
    output A_less
);

    // Extend both 4-bit inputs to 5 bits to capture any borrow from the subtraction.
    wire [4:0] diff;
    assign diff = {1'b0, A} - {1'b0, B};

    // The subtraction yields:
    // - No borrow (diff[4] == 0) and non-zero result => A is greater than B.
    // - No borrow (diff[4] == 0) and zero result  => A equals B.
    // - Borrow (diff[4] == 1)                      => A is less than B.
    assign A_equal   = (diff[4] == 1'b0) && (diff[3:0] == 4'b0000);
    assign A_greater = (diff[4] == 1'b0) && (diff[3:0] != 4'b0000);
    assign A_less    = (diff[4] == 1'b1);

endmodule
