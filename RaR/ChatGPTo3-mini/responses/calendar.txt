module calendar(
    input CLK,
    input RST,
    output reg [5:0] Secs,
    output reg [5:0] Mins,
    output reg [5:0] Hours
);

  // Seconds Handling Block
  always @(posedge CLK or posedge RST) begin
    if (RST)
      Secs <= 6'd0;
    else if (Secs == 6'd59)
      Secs <= 6'd0;
    else
      Secs <= Secs + 6'd1;
  end
  
  // Minutes Handling Block
  always @(posedge CLK or posedge RST) begin
    if (RST)
      Mins <= 6'd0;
    // Trigger on seconds counter finishing its cycle.
    else if (Secs == 6'd59) begin
      if (Mins == 6'd59)
        Mins <= 6'd0;
      else
        Mins <= Mins + 6'd1;
    end
    else
      Mins <= Mins;  // Hold the current value
  end
  
  // Hours Handling Block
  always @(posedge CLK or posedge RST) begin
    if (RST)
      Hours <= 6'd0;
    // Increment Hours only when both Mins and Secs have reached their max (59)
    else if ((Secs == 6'd59) && (Mins == 6'd59)) begin
      if (Hours == 6'd23)
        Hours <= 6'd0;
      else
        Hours <= Hours + 6'd1;
    end
    else
      Hours <= Hours;  // Hold the current value
  end

endmodule
